Maths & Equality

```js
var x = 2;

var z = 4;

console.log(x == x);
console.log(x == y);
console.log(x == a);
console.log(a == b);
console.log(x === y);
console.log(x === a);
console.log(x !== a);
console.log(x !== z);

// When you see console.assert fill in the _ to make the statement true
console.assert(x _ y === -1);
console.assert(x _ y === 5);
console.assert(z _ x === 2);
```
