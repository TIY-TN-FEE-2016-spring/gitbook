### Functions and Fat Arrow Syntax

```js
var x = function() {
  console.log('hello');
};

function y() {
  console.log('world');
}

var z = () => {
  console.log('yup');
};

var a = () => {
  return 'no';
};


console.log(x); // function() { console.log('hello') }
console.log(x()); // "hello" // undefined
console.log(y()); // "world" // undefined
console.log(z()); // "yup" // undefined
console.log(a()); // "no"
```
